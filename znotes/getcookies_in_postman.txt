postman will automaticall detect the cookies

const token = req.cookies.jwt_token;

// cookie expires 

const options = {
    expires: new Date(Date.now() + days<5> * hours<24> * minutes<60> * seconds<60> * milliseconds<1000>)
    httpOnly: true, // only the server can change the cookie - user cannot directly change from browser
}

res.cookie("token", token, options);
